// 1. استقبال البيانات وتحديث الحالات (POST Request) function doPost(e) { try { var sheet = getOrCreateTelemetrySheet(); var timestamp = new Date(); // قراءة البيانات من الرابط أو الجسم var action = e.parameter.action; var empId = e.parameter.emp_id; var data = null; if (e.postData && e.postData.contents) { try { data = JSON.parse(e.postData.contents); if (data) { if (!action) action = data.action; if (!empId) empId = data.emp_id; } } catch (err) {} } // أ) معالجة إنهاء حالة الطوارئ (Resolve) if (action === "resolve" && empId) { return handleResolveEmergency(sheet, empId.toString(), timestamp); } // ب) معالجة بيانات التتبع المرسلة من الساعة if (!data) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": "No telemetry data received" })) .setMimeType(ContentService.MimeType.JSON); } var watchEmpId = (data.emp_id || "WATCH_001").toString().trim().toUpperCase(); // جلب الاسم المطابق لرمز الساعة من جدول الموظفين (مثال: WATCH_001 -> أحمد منصور) var empName = getEmployeeNameFromDatabase(watchEmpId); // استخدام محلل الأرقام الآمن لتجاوز الفواصل العربية (٫) var lat = safeParseFloat(data.lat, 24.7136); var lng = safeParseFloat(data.lng, 46.6753); var speed = data.speed !== undefined ? data.speed.toString() : "0.0"; var hr = data.heart_rate !== undefined ? parseInt(data.heart_rate) : 0; var steps = data.steps !== undefined ? parseInt(data.steps) : 0; var battery = typeof data.battery !== 'undefined' ? data.battery.toString().replace('%', '') + "%" : "100%"; var isWorn = data.is_worn ? "YES" : "NO"; // توحيد الحالات var rawStatus = (data.status || "NORMAL").toUpperCase(); var status = "NORMAL"; if (rawStatus === "SOS" || rawStatus === "EMERGENCY") { status = "SOS"; } else if (rawStatus === "WARNING") { status = "WARNING"; } // جلب آخر حالة للموظف في جدول التتبع var currentStatus = "NORMAL"; var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][1].toString().trim().toUpperCase() === watchEmpId) { currentStatus = rows[i][10].toString().trim().toUpperCase(); break; } } // قفل الأمان (Safety Lock): إذا كان مسجلاً في الطوارئ حالياً، نحميه من إيقاف الإنذار تلقائياً if (currentStatus === "SOS" && status !== "SOS") { status = "SOS"; } // حساب أقرب مسعف متاح وكتابته مباشرة في صف الطوارئ var nearestResponder = "لا يوجد"; if (status === "SOS") { nearestResponder = findNearestResponderInSheet(sheet, watchEmpId, lat, lng); } // إضافة صف التتبع الفعلي sheet.appendRow([ timestamp, watchEmpId, empName, lat, lng, speed, hr, steps, battery, isWorn, status, nearestResponder ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "status": "recorded", "dispatch_to": nearestResponder })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // 2. جلب وتصدير البيانات وإنهاء الحالة (GET Request) function doGet(e) { try { var sheet = getOrCreateTelemetrySheet(); var timestamp = new Date(); var action = e.parameter.action; var empId = e.parameter.emp_id; // أ) معالجة إنهاء حالة الطوارئ عبر GET if (action === "resolve" && empId) { return handleResolveEmergency(sheet, empId.toString(), timestamp); } // ب) جلب أحدث الحالات للوحة المراقبة var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } var headers = rows[0]; var headerMap = {}; for (var i = 0; i < headers.length; i++) { headerMap[headers[i].toString().trim().toLowerCase()] = i; } var latestRecords = {}; for (var r = rows.length - 1; r >= 1; r--) { var row = rows[r]; var empIdIdx = headerMap["employee id"]; if (empIdIdx === undefined || empIdIdx >= row.length) continue; var id = row[empIdIdx].toString().trim().toUpperCase(); if (id === "") continue; if (!latestRecords[id]) { var getVal = function(headerName, defaultVal) { var idx = headerMap[headerName.toLowerCase()]; if (idx !== undefined && idx < row.length) { var val = row[idx]; return (val !== null && val !== undefined && val !== "") ? val : defaultVal; } return defaultVal; }; var timeVal = getVal("Timestamp", ""); if (timeVal instanceof Date) { timeVal = timeVal.toISOString(); } latestRecords[id] = { timestamp: timeVal, emp_id: id, emp_name: getVal("Employee Name", "موظف غير معروف"), lat: safeParseFloat(getVal("Latitude", 24.7136), 24.7136), lng: safeParseFloat(getVal("Longitude", 46.6753), 46.6753), speed: getVal("Speed (m/s)", "0.0").toString(), heart_rate: parseInt(getVal("Heart Rate (BPM)", 0)) || 0, steps: parseInt(getVal("Steps", 0)) || 0, battery: getVal("Battery", "100%").toString(), is_worn: getVal("Is Worn", "YES").toString(), status: getVal("Status", "NORMAL").toString().toUpperCase(), nearest_responder: getVal("Nearest Responder", "لا يوجد").toString() }; } } var data = Object.keys(latestRecords).map(function(key) { return latestRecords[key]; }); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // دالة مساعدة لتحويل الفواصل العشرية بشكل آمن وتفادي الكومة العربية (٫) function safeParseFloat(val, defaultVal) { if (val === undefined || val === null || val === "") return defaultVal; var str = val.toString().replace(/٫/g, ".").replace(/,/g, "."); var num = parseFloat(str); return isNaN(num) ? defaultVal : num; } // دالة مساعدة لإنهاء حالة الطوارئ وإضافة صف NORMAL function handleResolveEmergency(sheet, empId, timestamp) { var empName = getEmployeeNameFromDatabase(empId); sheet.appendRow([ timestamp, empId, empName, 24.7136, 46.6753, "0.0", 75, 0, "100%", "YES", "NORMAL", "لا يوجد" ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "message": "Emergency resolved successfully for " + empName })) .setMimeType(ContentService.MimeType.JSON); } // دالة للبحث عن اسم الموظف بالاعتماد على رقمه الوظيفي من جدول الموظفين function getEmployeeNameFromDatabase(empId) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dbSheet = getOrCreateEmployeesSheet(ss); var rows = dbSheet.getDataRange().getValues(); var cleanId = empId.toString().trim().toUpperCase(); for (var i = 1; i < rows.length; i++) { if (rows[i][0].toString().trim().toUpperCase() === cleanId) { return rows[i][1]; } } return "حارس " + empId; } // دالة جلب أو إنشاء ورقة الموظفين وتعبئتها تلقائياً function getOrCreateEmployeesSheet(ss) { var sheet = ss.getSheetByName("Employees"); var rebuild = false; if (sheet) { var firstValue = sheet.getRange(2, 1).getValue().toString().trim(); if (firstValue === "101") { rebuild = true; sheet.clearContents(); } } else { sheet = ss.insertSheet("Employees"); rebuild = true; } if (rebuild) { sheet.appendRow(["Employee ID", "Employee Name", "Role"]); var employees = [ ["WATCH_001", "أحمد منصور", "مشرف ميداني"], ["WATCH_002", "خالد الحربي", "حارس أمن"], ["WATCH_003", "عبد الرحمن العتيبي", "حارس أمن"], ["WATCH_004", "محمد القحطاني", "حارس أمن"], ["WATCH_005", "سعد الدوسري", "مشرف دورية"], ["WATCH_006", "فيصل المطيري", "حارس بوابات"], ["WATCH_007", "سلطان الشمري", "حارس أمن"], ["WATCH_008", "عبد الله الغامدي", "حارس أمن"], ["WATCH_009", "عمر الزهراني", "حارس أمن"], ["WATCH_010", "فهد المالكي", "حارس أمن"], ["WATCH_011", "سلمان الشهري", "مسعف ميداني"], ["WATCH_012", "بدر العنزي", "حارس أمن"], ["WATCH_013", "يوسف الرشيدي", "حارس أمن"], ["WATCH_014", "خليل السبيعي", "حارس أمن"], ["WATCH_015", "صالح الدوسري", "حارس أمن"] ]; for (var i = 0; i < employees.length; i++) { sheet.appendRow(employees[i]); } var headerRange = sheet.getRange(1, 1, 1, 3); headerRange.setBackground("#3B82F6").setFontColor("#FFFFFF").setFontWeight("bold"); sheet.setFrozenRows(1); } return sheet; } // دالة جلب أو إنشاء ورقة التتبع وإعداد التنسيقات function getOrCreateTelemetrySheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Telemetry"); if (!sheet) { sheet = ss.insertSheet("Telemetry"); var headers = [ "Timestamp", "Employee ID", "Employee Name", "Latitude", "Longitude", "Speed (m/s)", "Heart Rate (BPM)", "Steps", "Battery", "Is Worn", "Status", "Nearest Responder" ]; sheet.appendRow(headers); var headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setBackground("#1F2937") .setFontColor("#FFFFFF") .setFontWeight("bold") .setHorizontalAlignment("center"); sheet.setFrozenRows(1); var statusRange = sheet.getRange("K2:K1000"); var ruleNormal = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("NORMAL") .setBackground("#D1FAE5") .setFontColor("#065F46") .setBold(true) .setRanges([statusRange]) .build(); var ruleWarning = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("WARNING") .setBackground("#FEF3C7") .setFontColor("#92400E") .setBold(true) .setRanges([statusRange]) .build(); var ruleSos = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("SOS") .setBackground("#FEE2E2") .setFontColor("#991B1B") .setBold(true) .setRanges([statusRange]) .build(); var rules = sheet.getConditionalFormatRules(); rules.push(ruleNormal, ruleWarning, ruleSos); sheet.setConditionalFormatRules(rules); } return sheet; } // حساب أقرب مستجيب في الإكسيل function findNearestResponderInSheet(sheet, sosEmpId, sosLat, sosLng) { if (sosLat === 0 && sosLng === 0) return "لا يوجد"; var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return "لا يوجد"; var headers = rows[0]; var headerMap = {}; for (var i = 0; i < headers.length; i++) { headerMap[headers[i].toString().trim().toLowerCase()] = i; } var latestEmployees = {}; for (var r = rows.length - 1; r >= 1; r--) { var row = rows[r]; var empId = row[headerMap["employee id"]].toString().trim().toUpperCase(); if (empId === "" || empId === sosEmpId.toUpperCase()) continue; if (!latestEmployees[empId]) { var getVal = function(colName) { var idx = headerMap[colName.toLowerCase()]; return (idx !== undefined && idx < row.length) ? row[idx] : ""; }; latestEmployees[empId] = { emp_name: getVal("Employee Name"), lat: safeParseFloat(getVal("Latitude"), 0), lng: safeParseFloat(getVal("Longitude"), 0), status: getVal("Status").toString().toUpperCase(), timestamp: new Date(getVal("Timestamp")) }; } } var nearestName = "لا يوجد مستجيب متاح"; var minDistance = Infinity; for (var empId in latestEmployees) { var emp = latestEmployees[empId]; if (emp.status === "SOS") continue; if (isNaN(emp.timestamp.getTime()) || (new Date() - emp.timestamp) > 300000) continue; if (emp.lat === 0 || emp.lng === 0) continue; var dist = calculateHaversineDistance(sosLat, sosLng, emp.lat, emp.lng); if (dist < minDistance) { minDistance = dist; var distStr = dist < 1000 ? Math.round(dist) + " متر" : (dist / 1000).toFixed(1) + " كم"; nearestName = emp.emp_name + " (" + empId + ") · على بعد " + distStr; } } return nearestName; } function calculateHaversineDistance(lat1, lon1, lat2, lon2) { var R = 6371e3; var phi1 = lat1 * Math.PI / 180; var phi2 = lat2 * Math.PI / 180; var deltaPhi = (lat2 - lat1) * Math.PI / 180; var deltaLambda = (lon2 - lon1) * Math.PI / 180; var a = Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) + Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2); var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); return R * c; }
