// 1. استقبال البيانات وتحديث الحالات (POST Request) function doPost(e) { try { var sheet = getOrCreateTelemetrySheet(); var timestamp = new Date(); // قراءة المعاملات من الرابط (Query Parameters) أو من جسم الطلب (JSON Body) var action = e.parameter.action; var empId = e.parameter.emp_id; var data = null; if (e.postData && e.postData.contents) { try { data = JSON.parse(e.postData.contents); if (data) { if (!action) action = data.action; if (!empId) empId = data.emp_id; } } catch (err) {} } // أ) معالجة إنهاء حالة الطوارئ (Resolve) المرسلة من لوحة التحكم if (action === "resolve" && empId) { empId = empId.toString(); var empName = "موظف"; // جلب اسم الموظف الفعلي من سجلاته السابقة var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][1].toString() === empId) { empName = rows[i][2]; break; } } // إضافة سجل إغلاق حالة الطوارئ sheet.appendRow([ timestamp, empId, empName, 24.7136, 46.6753, "0.0", 75, 0, "100%", "YES", "NORMAL", "لا يوجد" ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "message": "Emergency resolved successfully" })) .setMimeType(ContentService.MimeType.JSON); } // ب) معالجة بيانات التتبع المرسلة من ساعة الموظف if (!data) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": "No telemetry data received" })) .setMimeType(ContentService.MimeType.JSON); } var watchEmpId = (data.emp_id || "EMP-001").toString(); var empName = data.emp_name || "موظف غير معروف"; var lat = Number.isFinite(parseFloat(data.lat)) ? parseFloat(data.lat) : 24.7136; var lng = Number.isFinite(parseFloat(data.lng)) ? parseFloat(data.lng) : 46.6753; var speed = data.speed !== undefined ? data.speed.toString() : "0.0"; var hr = data.heart_rate !== undefined ? parseInt(data.heart_rate) : 0; var steps = data.steps !== undefined ? parseInt(data.steps) : 0; var battery = typeof data.battery !== 'undefined' ? data.battery.toString().replace('%', '') + "%" : "100%"; var isWorn = data.is_worn ? "YES" : "NO"; var status = (data.status || "NORMAL").toUpperCase(); // حساب أقرب مسعف متاح وكتابته مباشرة في صف الطوارئ var nearestResponder = "لا يوجد"; if (status === "SOS") { nearestResponder = findNearestResponderInSheet(sheet, watchEmpId, lat, lng); } // إضافة صف التتبع الفعلي sheet.appendRow([ timestamp, watchEmpId, empName, lat, lng, speed, hr, steps, battery, isWorn, status, nearestResponder ]); return ContentService.createTextOutput(JSON.stringify({ "result": "success", "status": "recorded", "dispatch_to": nearestResponder })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // 2. جلب وتصدير البيانات الحية للوحة التحكم (GET Request) function doGet(e) { try { var sheet = getOrCreateTelemetrySheet(); var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } var headers = rows[0]; var headerMap = {}; for (var i = 0; i < headers.length; i++) { headerMap[headers[i].toString().trim().toLowerCase()] = i; } var latestRecords = {}; for (var r = rows.length - 1; r >= 1; r--) { var row = rows[r]; var empIdIdx = headerMap["employee id"]; if (empIdIdx === undefined || empIdIdx >= row.length) continue; var empId = row[empIdIdx].toString().trim(); if (empId === "") continue; if (!latestRecords[empId]) { var getVal = function(headerName, defaultVal) { var idx = headerMap[headerName.toLowerCase()]; if (idx !== undefined && idx < row.length) { var val = row[idx]; return (val !== null && val !== undefined && val !== "") ? val : defaultVal; } return defaultVal; }; var timestamp = getVal("Timestamp", ""); if (timestamp instanceof Date) { timestamp = timestamp.toISOString(); } latestRecords[empId] = { timestamp: timestamp, emp_id: empId, emp_name: getVal("Employee Name", "موظف غير معروف"), lat: parseFloat(getVal("Latitude", 24.7136)) || 24.7136, lng: parseFloat(getVal("Longitude", 46.6753)) || 46.6753, speed: getVal("Speed (m/s)", "0.0").toString(), heart_rate: parseInt(getVal("Heart Rate (BPM)", 0)) || 0, steps: parseInt(getVal("Steps", 0)) || 0, battery: getVal("Battery", "100%").toString(), is_worn: getVal("Is Worn", "YES").toString(), status: getVal("Status", "NORMAL").toString().toUpperCase(), nearest_responder: getVal("Nearest Responder", "لا يوجد").toString() }; } } var data = Object.keys(latestRecords).map(function(key) { return latestRecords[key]; }); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "result": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // 3. دالة جلب أو إنشاء ورقة التتبع وإعداد الأعمدة والتنسيقات function getOrCreateTelemetrySheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Telemetry"); if (!sheet) { sheet = ss.insertSheet("Telemetry"); var headers = [ "Timestamp", "Employee ID", "Employee Name", "Latitude", "Longitude", "Speed (m/s)", "Heart Rate (BPM)", "Steps", "Battery", "Is Worn", "Status", "Nearest Responder" ]; sheet.appendRow(headers); var headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setBackground("#1F2937") .setFontColor("#FFFFFF") .setFontWeight("bold") .setHorizontalAlignment("center"); sheet.setFrozenRows(1); var statusRange = sheet.getRange("K2:K1000"); var ruleNormal = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("NORMAL") .setBackground("#D1FAE5") .setFontColor("#065F46") .setBold(true) .setRanges([statusRange]) .build(); var ruleWarning = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("WARNING") .setBackground("#FEF3C7") .setFontColor("#92400E") .setBold(true) .setRanges([statusRange]) .build(); var ruleSos = SpreadsheetApp.newConditionalFormatRule() .whenTextEqualTo("SOS") .setBackground("#FEE2E2") .setFontColor("#991B1B") .setBold(true) .setRanges([statusRange]) .build(); var rules = sheet.getConditionalFormatRules(); rules.push(ruleNormal, ruleWarning, ruleSos); sheet.setConditionalFormatRules(rules); var defaultSheet = ss.getSheetByName("Sheet1"); if (defaultSheet && defaultSheet.getLastRow() === 0) { try { ss.deleteSheet(defaultSheet); } catch (err) {} } } return sheet; } // 4. خوارزمية هافرسين لحساب أقرب حارس أمن متصل جغرافياً function findNearestResponderInSheet(sheet, sosEmpId, sosLat, sosLng) { if (sosLat === 0 && sosLng === 0) return "لا يوجد"; var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return "لا يوجد"; var headers = rows[0]; var headerMap = {}; for (var i = 0; i < headers.length; i++) { headerMap[headers[i].toString().trim().toLowerCase()] = i; } var latestEmployees = {}; for (var r = rows.length - 1; r >= 1; r--) { var row = rows[r]; var empId = row[headerMap["employee id"]].toString().trim(); if (empId === "" || empId === sosEmpId) continue; if (!latestEmployees[empId]) { var getVal = function(colName) { var idx = headerMap[colName.toLowerCase()]; return (idx !== undefined && idx < row.length) ? row[idx] : ""; }; latestEmployees[empId] = { emp_name: getVal("Employee Name"), lat: parseFloat(getVal("Latitude")), lng: parseFloat(getVal("Longitude")), status: getVal("Status").toString().toUpperCase(), timestamp: new Date(getVal("Timestamp")) }; } } var nearestName = "لا يوجد مستجيب متاح"; var minDistance = Infinity; for (var empId in latestEmployees) { var emp = latestEmployees[empId]; if (emp.status === "SOS") continue; // استبعاد الموظفين غير المتصلين (أكثر من 5 دقائق بلا تحديث) if (isNaN(emp.timestamp.getTime()) || (new Date() - emp.timestamp) > 300000) continue; if (isNaN(emp.lat) || isNaN(emp.lng)) continue; var dist = calculateHaversineDistance(sosLat, sosLng, emp.lat, emp.lng); if (dist < minDistance) { minDistance = dist; var distStr = dist < 1000 ? Math.round(dist) + " متر" : (dist / 1000).toFixed(1) + " كم"; nearestName = emp.emp_name + " (" + empId + ") · على بعد " + distStr; } } return nearestName; } function calculateHaversineDistance(lat1, lon1, lat2, lon2) { var R = 6371e3; // بالمتر var phi1 = lat1 * Math.PI / 180; var phi2 = lat2 * Math.PI / 180; var deltaPhi = (lat2 - lat1) * Math.PI / 180; var deltaLambda = (lon2 - lon1) * Math.PI / 180; var a = Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) + Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2); var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); return R * c; }
